function solution(word) {
    const chars = ['A', 'E', 'I', 'O', 'U'];
    const tmp = Array(5).fill(null);
    let count = 0;
    let answer;
    function DFS(L) {
        if (tmp.join('') === word) {
            answer = count;
            return;
        }
        if (L === 5) {
            return;
        } else {
            for (let i = 0; i < 5; i++) {
                tmp[L] = chars[i];
                count += 1;
                DFS(L + 1, count + 1);
                tmp[L] = null;
            }
        }
    }
    DFS(0);
    return answer;
}
