let majorityElement = function (nums) {
    let dict = {};
    nums.forEach((ele) => {
        if (dict[ele]) dict[ele]++;
        else dict[ele] = 1;
    });
    dict = Object.entries(dict).sort((a, b) => a[1] - b[1]);
    return parseInt(dict.pop()[0]);
};
